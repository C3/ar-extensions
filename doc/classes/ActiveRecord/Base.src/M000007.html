  def self.validations_array_for_column_names_and_attributes( column_names, array_of_attributes )
    arr = []
    array_of_attributes.each do |attributes|
      c = 0
      hsh = attributes.inject( {} ){|hsh,attr| hsh[ column_names[c] ] = attr ; c+=1 ; hsh }
      arr << hsh
    end
    arr
  end
