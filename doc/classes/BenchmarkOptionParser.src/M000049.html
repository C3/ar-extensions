  def self.print_options( options )
    puts "Benchmarking the following options:"
    puts "\tDatabase adapter: #{options.adapter}"
    puts "\tNumber of objects: #{options.number_of_objects}"
    puts "\tTable types:"
    print_valid_table_types( options, :prefix=>"\t\t" )
  end
