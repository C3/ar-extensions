  def bm( description, &blk )
    tms  = nil
    puts "Benchmarking #{description}"
    
    begin
      Benchmark.bm { |x| tms = x.report { blk.call } }
      failed = false
    rescue Exception => ex
      failed = true
    end
    OpenStruct.new :description=>description, :tms=>tms, :failed=>failed
  end
